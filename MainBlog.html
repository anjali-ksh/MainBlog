import streamlit as st
import pandas as pd
from PIL import Image
from sklearn.datasets import load_iris
from sklearn.ensemble import RandomForestClassifier

st.set_page_config(page_title="Streamlit Demo", layout="centered")
st.title("ğŸŒŸ Streamlit App: Upload Files + ML Prediction")

st.sidebar.header("Choose Action")
option = st.sidebar.radio("Select a section:", ["ğŸ“· Upload Image", "ğŸ“„ Upload CSV", "ğŸ¤– Predict Iris Species"])

if option == "ğŸ“· Upload Image":
    st.subheader("Upload an Image")
    image_file = st.file_uploader("Choose an image file", type=["jpg", "png", "jpeg"])
    if image_file:
        img = Image.open(image_file)
        st.image(img, caption="Uploaded Image", use_column_width=True)

elif option == "ğŸ“„ Upload CSV":
    st.subheader("Upload a CSV File")
    csv_file = st.file_uploader("Choose a CSV file", type="csv")
    if csv_file:
        df = pd.read_csv(csv_file)
        st.write("CSV Preview:")
        st.dataframe(df.head())

elif option == "ğŸ¤– Predict Iris Species":
    st.subheader("ML Prediction: Iris Flower")
    iris = load_iris()
    X = iris.data
    y = iris.target
    model = RandomForestClassifier()
    model.fit(X, y)

    sl = st.slider("Sepal Length (cm)", 4.0, 8.0, 5.1)
    sw = st.slider("Sepal Width (cm)", 2.0, 4.5, 3.5)
    pl = st.slider("Petal Length (cm)", 1.0, 7.0, 1.4)
    pw = st.slider("Petal Width (cm)", 0.1, 2.5, 0.2)

    input_data = [[sl, sw, pl, pw]]
    prediction = model.predict(input_data)[0]
    predicted_species = iris.target_names[prediction]

    st.success(f"ğŸŒ¼ Predicted Iris Species: {predicted_species.capitalize()}")
